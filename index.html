    };

    

    setContentIdea(ideaWithEnhancements);

    setGeneratingContent(false);

  };


  const markAsReady = () => {

    if (contentIdea && contentIdea.id) {

      setContentIdeasDB(prev => prev.map(idea => 

        idea.id === contentIdea.id 

          ? { ...idea, status: "Ready" }

          : idea

      ));

      

      setTimeout(() => {

        alert(language === 'en' ? 'Marked as Ready! ✅' : language === 'th' ? 'ทำเครื่องหมายเป็นพร้อมแล้ว! ✅' : 'Отмечено как готово! ✅');

      }, 100);

    }

  };


  const markAsBullshit = () => {

    if (contentIdea && contentIdea.id) {

      setContentIdeasDB(prev => prev.map(idea => 

        idea.id === contentIdea.id 

          ? { ...idea, status: "Bullshit" }

          : idea

      ));

      

      setTimeout(() => {

        alert(language === 'en' ? 'Marked as Bullshit! 💩' : language === 'th' ? 'ทำเครื่องหมายเป็นเหลวไหล! 💩' : 'Отмечено как фигня! 💩');

      }, 100);

    }

  };


  const handleAddStrain = () => {

    if (!newStrain.name || !newStrain.price) {

      alert(t.fillAllFields);

      return;

    }

    

    const strain = {

      ...newStrain,

      id: Date.now(),

      date: new Date().toLocaleDateString()

    };

    

    setStrains([...strains, strain]);

    setNewStrain({ type: 'indica', name: '', price: '' });

    setShowAddForm(false);

    

    setTimeout(() => alert(t.strainAdded), 100);

  };


  return (

    <div className="min-h-screen bg-black text-white font-sans">

      {/* Header */}

      <div className="bg-white text-black p-6 text-center relative">

        {/* Home/Refresh button on the left */}

        <div className="absolute top-4 left-4">

          <button

            onClick={() => window.location.reload()}

            className="p-2 rounded bg-gray-200 hover:bg-gray-300 transition-colors"

            title="Refresh Page"

          >

            🏠

          </button>

        </div>

        

        <div className="absolute top-4 right-4 flex items-center space-x-2">

          <button

            onClick={() => setLanguage('en')}

            className={`p-2 rounded ${language === 'en' ? 'bg-black text-white' : 'bg-gray-200'}`}

          >

            🇺🇸

          </button>

          <button

            onClick={() => setLanguage('th')}

            className={`p-2 rounded ${language === 'th' ? 'bg-black text-white' : 'bg-gray-200'}`}

          >

            🇹🇭

          </button>

          <button

            onClick={() => setLanguage('ru')}

            className={`p-2 rounded ${language === 'ru' ? 'bg-black text-white' : 'bg-gray-200'}`}

          >

            🇷🇺

          </button>

        </div>

        <h1 className="text-4xl font-bold mb-2">{t.title}</h1>

        <p className="text-xl text-gray-600">{t.subtitle}</p>

      </div>


      {/* Main Content */}

      <div className="p-6">

        {/* Main Menu */}

        {!showAddForm && !showQRCodes && (

          <div className="space-y-6">

            {/* Content Generator */}

            <div className="bg-white text-black rounded-lg p-6 text-center">

              <button

                onClick={generateContentIdea}

                disabled={generatingContent}

                className="w-full bg-gradient-to-br from-purple-600 via-pink-500 to-orange-400 text-white p-6 rounded-xl text-xl font-bold hover:from-purple-700 hover:via-pink-600 hover:to-orange-500 transition-all duration-300 disabled:opacity-50 flex items-center justify-center space-x-3 shadow-lg transform hover:scale-105 active:scale-95"

              >

                <MessageCircle size={32} />

                <span>{generatingContent ? t.generating : t.contentGenerator}</span>

              </button>

              

              {contentIdea && (

                <div className="mt-4 bg-gradient-to-br from-purple-500 via-pink-500 to-red-500 p-1 rounded-2xl">

                  <div className="bg-black rounded-2xl p-6 text-white text-center">

                    

                    {/* Goal at the top */}

                    <div className="mb-6">

                      <div className="text-yellow-300 text-sm font-bold mb-2">🎯 {t.goal}</div>

                      <div className="text-lg font-bold text-white">

                        {translateContent(contentIdea.goal, language)}

                      </div>

                    </div>


                    {/* Idea description in the middle */}

                    <div className="mb-6">

                      <div className="text-blue-300 text-sm font-bold mb-2">💡 {t.idea}</div>

                      <div className="text-base text-gray-300 leading-relaxed mb-3">

                        {translateContent(contentIdea.idea, language)}

                      </div>

                      {/* Implementation tip */}

                      {contentIdea.implementationTip && (

                        <div className="text-xs text-gray-400 italic border-l-2 border-gray-600 pl-3">

                          {translateContent(contentIdea.implementationTip, language)}

                        </div>

                      )}

                    </div>


                    {/* Post text at the bottom */}

                    <div className="mb-6">

                      <div className="text-green-300 text-sm font-bold mb-2">📝 {t.postText}</div>

                      <div className="text-xl font-bold text-white mb-3">

                        {contentIdea.exampleText}

                      </div>

                      

                      {/* Tags */}

                      <div className="flex flex-wrap gap-1 justify-center">

                        {contentIdea.tags.split(' ').map((tag, index) => (

                          <button

                            key={index}

                            onClick={() => {

                              navigator.clipboard.writeText(tag);

                              // Show brief visual feedback

                              const button = document.activeElement;

                              const originalText = button.textContent;

                              button.textContent = '✓';

                              button.style.backgroundColor = '#22c55e';

                              setTimeout(() => {

                                button.textContent = originalText;

                                button.style.backgroundColor = '';

                              }, 500);

                            }}

                            className="bg-gray-800 hover:bg-gray-700 px-2 py-1 rounded-full text-xs text-blue-300 hover:text-blue-200 transition-all cursor-pointer active:scale-95"

                            title={`Click to copy: ${tag}`}

                          >

                            {tag}

                          </button>

                        ))}

                      </div>

                    </div>


                    {/* Action Buttons */}

                    <div className="space-y-3">

                      <button

                        onClick={() => navigator.clipboard.writeText(`${contentIdea.exampleText}\n\n${contentIdea.tags}`)}

                        className="w-full bg-gradient-to-r from-purple-500 to-pink-500 text-white px-4 py-3 rounded-full text-sm font-bold hover:from-purple-600 hover:to-pink-600 transition-all"

                      >

                        📋 {t.copyPostText}

                      </button>

                      

                      <div className="flex space-x-3">

                        <button

                          onClick={markAsReady}

                          className="flex-1 bg-gradient-to-r from-green-600 to-green-500 text-white px-4 py-3 rounded-full text-sm font-bold hover:from-green-700 hover:to-green-600 transition-all"

                        >

                          ✅ {t.markAsReady}

                        </button>

                        <button

                          onClick={markAsBullshit}

                          className="flex-1 bg-gradient-to-r from-red-600 to-red-500 text-white px-4 py-3 rounded-full text-sm font-bold hover:from-red-700 hover:to-red-600 transition-all"

                        >

                          💩 {t.markAsBullshit}

                        </button>

                      </div>

                    </div>


                  </div>

                </div>

              )}

            </div>


            {/* Add New Weed */}

            <div className="bg-white text-black rounded-lg p-6 text-center">

              <button

                onClick={() => setShowAddForm(true)}

                className="w-full bg-gradient-to-r from-green-500 to-green-600 text-white p-6 rounded-lg text-xl font-bold hover:from-green-600 hover:to-green-700 transition-colors flex items-center justify-center space-x-3"

              >

                <Plus size={32} />

                <span>{t.addNewWeed}</span>

              </button>

            </div>


            {/* QR Codes */}

            <div className="bg-white text-black rounded-lg p-6 text-center">

              <button

                onClick={() => setShowQRCodes(true)}

                className="w-full bg-black text-white p-6 rounded-lg text-xl font-bold hover:bg-gray-800 transition-colors flex items-center justify-center space-x-3"

              >

                <QrCode size={32} />

                <span>{t.showQRCodes}</span>

              </button>

            </div>


            {/* Strains List */}

            {strains.length > 0 && (

              <div className="bg-white text-black rounded-lg p-6">

                <h3 className="text-xl font-bold mb-4">Recent Strains:</h3>

                <div className="space-y-3">

                  {strains.slice(-3).map(strain => (

                    <div key={strain.id} className="border-b pb-2">

                      <div className="flex justify-between items-center">

                        <span className="font-bold">{strain.name}</span>

                        <span className="text-green-600">{strain.price} THB/g</span>

                      </div>

                      <span className="text-sm text-gray-600 capitalize">{strain.type}</span>

                    </div>

                  ))}

                </div>

              </div>

            )}

          </div>

        )}


        {/* Add Strain Form */}

        {showAddForm && (

          <div className="bg-white text-black rounded-lg p-6">

            <h2 className="text-2xl font-bold mb-6 text-center">{t.addNewWeed}</h2>

            

            <div className="space-y-4">

              <div>

                <label className="block text-lg font-bold mb-2">{t.weedType}</label>

                <select

                  value={newStrain.type}

                  onChange={(e) => setNewStrain({...newStrain, type: e.target.value})}

                  className="w-full p-4 text-lg border rounded-lg focus:outline-none focus:ring-2 focus:ring-black"

                >

                  <option value="indica">{t.indica}</option>

                  <option value="sativa">{t.sativa}</option>

                  <option value="hybrid">{t.hybrid}</option>

                </select>

              </div>


              <div>

                <label className="block text-lg font-bold mb-2">{t.strainName}</label>

                <input

                  type="text"

                  value={newStrain.name}

                  onChange={(e) => setNewStrain({...newStrain, name: e.target.value})}

                  className="w-full p-4 text-lg border rounded-lg focus:outline-none focus:ring-2 focus:ring-black"

                  placeholder="Enter strain name..."

                />

              </div>


              <div>

                <label className="block text-lg font-bold mb-2">{t.pricePerGram}</label>

                <input

                  type="number"

                  value={newStrain.price}

                  onChange={(e) => setNewStrain({...newStrain, price: e.target.value})}

                  className="w-full p-4 text-lg border rounded-lg focus:outline-none focus:ring-2 focus:ring-black"

                  placeholder="Enter price..."

                />

              </div>


              <div className="flex space-x-4 pt-4">

                <button

                  onClick={handleAddStrain}

                  className="flex-1 bg-black text-white p-4 rounded-lg text-xl font-bold hover:bg-gray-800 transition-colors"

                >

                  {t.save}

                </button>

                <button

                  onClick={() => setShowAddForm(false)}

                  className="flex-1 bg-gray-300 text-black p-4 rounded-lg text-xl font-bold hover:bg-gray-400 transition-colors"

                >

                  {t.cancel}

                </button>

              </div>

            </div>

          </div>

        )}


        {/* QR Codes */}

        {showQRCodes && (

          <div className="space-y-6">

            <div className="bg-white text-black rounded-lg p-6 text-center">

              <h2 className="text-2xl font-bold mb-6">{t.showQRCodes}</h2>

              

              <div className="space-y-6">

                {/* Instagram QR */}

                <div className="border rounded-lg p-4">

                  <div className="flex items-center justify-center space-x-3 mb-4">

                    <Instagram size={32} />

                    <h3 className="text-xl font-bold">{t.instagram}</h3>

                  </div>

                  <div className="bg-gray-100 p-8 rounded-lg mb-4">

                    <div className="w-32 h-32 bg-black mx-auto rounded-lg flex items-center justify-center">

                      <QrCode size={64} className="text-white" />

                    </div>

                  </div>

                  <p className="text-sm text-gray-600">{t.followForFreeJoint}</p>

                </div>


                {/* Google Maps QR */}

                <div className="border rounded-lg p-4">

                  <div className="flex items-center justify-center space-x-3 mb-4">

                    <MapPin size={32} />

                    <h3 className="text-xl font-bold">{t.googleMaps}</h3>

                  </div>

                  <div className="bg-gray-100 p-8 rounded-lg mb-4">

                    <div className="w-32 h-32 bg-black mx-auto rounded-lg flex items-center justify-center">

                      <QrCode size={64} className="text-white" />

                    </div>

                  </div>

                  <p className="text-sm text-gray-600">{t.reviewForFreeJoint}</p>

                </div>

              </div>


              <button

                onClick={() => setShowQRCodes(false)}

                className="w-full bg-black text-white p-4 rounded-lg text-xl font-bold hover:bg-gray-800 transition-colors mt-6"

              >

                {t.cancel}

              </button>

            </div>

          </div>

        )}

      </div>

    </div>

  );

};


export default BudsHelper;
